Installation:
    sudo apt update
    sudo apt install nginx

Location:
    --> /var/www/html - root location for default page, (use virtual host for multiple domain)
    --> create folder "sample" and create an html file
    --> /var/www/sample

Virtual Host:
    --> create "sample" file in /etc/nginx/sites-enabled/sample
    --> add this 
        server {
            listen 81;
            listen [::]:81;

            server_name YOUR_IP;

            root /var/www/sample;
            index index.html;

            location / {
                try_files $uri $uri/ =404;
            }
        }

Test:
    run:
        sudo service nginx restart or sudo systemctl reload nginx

    --> open url "http://192.168.1.37:81"
    --> it serve content in index.html

Check error-log:
    sudo tail -f /var/log/nginx/error.log
